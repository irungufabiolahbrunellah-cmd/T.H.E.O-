<script>

let badges = localStorage.getItem("theo2Badges") || 0;
badges = parseInt(badges);

const badgeDisplay = document.getElementById("badgeCount");
const progressBar = document.getElementById("progressBar");

badgeDisplay.textContent = badges;

updateProgress();
unlockThings();

function collectBadge(){

badges++;

localStorage.setItem("theo2Badges",badges);

badgeDisplay.textContent = badges;

document.getElementById("badgeSound").play();

updateProgress();

unlockThings();

}

function updateProgress(){

let percent = (badges/20)*100;

progressBar.style.width = percent+"%";

}

function unlockThings(){

if(badges>=3){
document.getElementById("friend1").style.display="block";
}

if(badges>=6){
document.getElementById("friend2").style.display="block";
}

if(badges>=10){
document.getElementById("friend3").style.display="block";
}

if(badges>=5){
document.getElementById("planet1").style.display="block";
}

if(badges>=10){
document.getElementById("planet2").style.display="block";
}

if(badges>=15){
document.getElementById("planet3").style.display="block";
}

if(badges>=1){
document.getElementById("ach1").textContent="✅ First Badge";
}

if(badges>=10){
document.getElementById("ach2").textContent="✅ Explorer";
}

if(badges>=20){
document.getElementById("ach3").textContent="✅ Space Master";
}

}

function resetGame(){

badges=0;

localStorage.setItem("theo2Badges",badges);

location.reload();

}

let canvas = document.getElementById("game");
let ctx = canvas.getContext("2d");

let theoX = 140;
let asteroidX = Math.random()*280;
let asteroidY = 0;

function startGame(){

setInterval(()=>{

ctx.clearRect(0,0,300,200);

ctx.font="30px Arial";
ctx.fillText("🧑‍🚀",theoX,170);

ctx.fillText("☄️",asteroidX,asteroidY);

asteroidY +=5;

if(asteroidY>200){
asteroidY=0;
asteroidX=Math.random()*280;
}

},50);

}
